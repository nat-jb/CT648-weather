CREATE TABLE weather (
    id SERIAL PRIMARY KEY,
    province  VARCHAR(255) NOT NULL,
    district  VARCHAR(255) NOT NULL, 
    temperature FLOAT NOT NULL,
    humidity FLOAT NOT NULL,
    wind_speed FLOAT NOT NULL,
    gusts FLOAT,
    condition TEXT,
    cloud_cover FLOAT,
    uv_index FLOAT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

const fetchAndStoreWeatherData = async () => {
    for (const LOCATION of locations) {
      try {
        // ดึงข้อมูลจาก WeatherAPI.com
        const response = await fetch(`http://api.weatherapi.com/v1/current.json?key=${API_KEY}&q=${LOCATION}`);
        const weatherData = await response.json();
  
        // เช็คว่ามีข้อมูลหรือไม่
        if (response.ok) {
          const { location: loc, current } = weatherData;
          const { temp_c, humidity, wind_kph, gust_kph, condition, cloud, uv } = current;
  
          // ตรวจสอบข้อมูลในฐานข้อมูล
          const query = 'SELECT * FROM weather WHERE location = $1 ORDER BY created_at DESC LIMIT 1';
          const result = await pool.query(query, [loc.name]);
  
          // เช็คว่ามีข้อมูลล่าสุดหรือไม่
          if (result.rows.length > 0) {
            const latestData = result.rows[0];
            // เปรียบเทียบข้อมูลใหม่กับข้อมูลล่าสุดในฐานข้อมูล
            if (
              latestData.temperature === temp_c &&
              latestData.humidity === humidity &&
              latestData.wind_speed === wind_kph &&
              latestData.gusts === gust_kph &&
              latestData.condition === condition.text &&
              latestData.cloud_cover === cloud &&
              latestData.uv_index === uv
            ) {
              console.log(`No changes in weather data for ${loc.name}. Skipping insert.`);
              continue; // ข้ามไปยังจังหวัดถัดไป
            }
          }
  
          // ถ้าข้อมูลมีการเปลี่ยนแปลง ให้ทำการบันทึกข้อมูลใหม่
          const insertQuery = `
            INSERT INTO weather (location, temperature, humidity, wind_speed, gusts, condition, cloud_cover, uv_index, created_at)
            VALUES ($1, $2, $3, $4, $5, $6, $7, $8, NOW()) RETURNING *`;
          const values = [loc.name, temp_c, humidity, wind_kph, gust_kph, condition.text, cloud, uv];
          await pool.query(insertQuery, values);
          
          console.log('Weather data inserted successfully:', { 
            location: loc.name, 
            temperature: temp_c, 
            humidity, 
            wind_speed: wind_kph,
            gusts: gust_kph,
            condition: condition.text,
            cloud_cover: cloud,
            uv_index: uv
          });
        } else {
          console.error('Error fetching weather data for', LOCATION, ':', weatherData);
        }
      } catch (error) {
        console.error('Error:', error);
      }
    }
  };

// ฟังก์ชันเพื่อตรวจสอบและจัดเก็บข้อมูลสภาพอากาศ
const fetchAndStoreWeatherData = async () => {
    for (const [province, subLocations] of Object.entries(locations)) {
        for (const district of subLocations) {
            try {
                // ดึงข้อมูลจาก WeatherAPI.com
                const response = await fetch(`http://api.weatherapi.com/v1/current.json?key=${API_KEY}&q=${encodeURIComponent(`${district}, ${province}`)}`);
                const weatherData = await response.json();

                // เช็คว่ามีข้อมูลหรือไม่
                if (response.ok) {
                    const { current } = weatherData;
                    const { temp_c, humidity, wind_kph, gust_kph, condition, cloud, uv } = current;

                    // ตรวจสอบข้อมูลในฐานข้อมูล
                    const query = 'SELECT * FROM weather WHERE province = $1 AND district = $2 ORDER BY created_at DESC LIMIT 1';
                    const result = await pool.query(query, [province, district]);

                    // เช็คว่ามีข้อมูลล่าสุดหรือไม่
                    if (result.rows.length > 0) {
                        const latestData = result.rows[0];
                        // เปรียบเทียบข้อมูลใหม่กับข้อมูลล่าสุดในฐานข้อมูล
                        if (
                            latestData.temperature === temp_c &&
                            latestData.humidity === humidity &&
                            latestData.wind_speed === wind_kph &&
                            latestData.gusts === gust_kph &&
                            latestData.condition === condition.text &&
                            latestData.cloud_cover === cloud &&
                            latestData.uv_index === uv
                        ) {
                            console.log(`No changes in weather data for ${district}, ${province}. Skipping insert.`);
                            continue; // ข้ามไปยังอำเภอต่อไป
                        }
                    }

                    // ถ้าข้อมูลมีการเปลี่ยนแปลง ให้ทำการบันทึกข้อมูลใหม่
                    const insertQuery = `INSERT INTO weather (province, district, temperature, humidity, wind_speed, gusts, condition, cloud_cover, uv_index, created_at) VALUES ($1, $2, $3, $4, $5, $6, $7, $8, $9, NOW())`;
                    await pool.query(insertQuery, [
                        province,
                        district,
                        temp_c,
                        humidity,
                        wind_kph,
                        gust_kph,
                        condition.text,
                        cloud,
                        uv
                    ]);
                    console.log(`Weather data for ${district} in ${province} updated successfully.`);
                } else {
                    console.error(`Error fetching weather data for ${district}, ${province}:`, weatherData.error);
                }
            } catch (error) {
                console.error('Error processing district:', district, error);
            }
        }
    }
};

  const [weatherHistory, setWeatherHistory] = useState([]);
  const [loadingWeather, setLoadingWeather] = useState(false);
  const [error, setError] = useState('');

  // Fetch provinces when the component mounts
  useEffect(() => {
    fetchProvinces();
  }, []);

  const fetchProvinces = async () => {
    try {
      const response = await fetch('http://localhost:5001/api/provinces');
      if (!response.ok) {
        throw new Error('Network response was not ok');
      }
      const data = await response.json();
      setProvinces(data);
    } catch (error) {
      setError('Error fetching provinces: ' + error.message);
    }
  };

  // Fetch districts when a province is selected
  useEffect(() => {
    if (selectedProvince) {
      fetchDistricts(selectedProvince);
    }
  }, [selectedProvince]);

  const fetchDistricts = async (province) => {
    try {
      const response = await fetch(`http://localhost:5001/api/districts/${province}`);
      if (!response.ok) {
        throw new Error('Network response was not ok');
      }
      const data = await response.json();
      setDistricts(data);
    } catch (error) {
      setError('Error fetching districts: ' + error.message);
    }
  };

  // Fetch weather history when a district is selected
  useEffect(() => {
    if (selectedDistrict) {
      fetchWeatherHistory(selectedDistrict);
    }
  }, [selectedDistrict]);

  const fetchWeatherHistory = async (district) => {
    setLoadingWeather(true);
    try {
      const response = await fetch(`http://localhost:5001/api/weather_history/${district}`);
      if (!response.ok) {
        throw new Error('Network response was not ok');
      }
      const data = await response.json();
      setWeatherHistory(data);
    } catch (error) {
      setError('Error fetching weather history: ' + error.message);
    } finally {
      setLoadingWeather(false);
    }
  };

  return (
    <div className="App">
      <h1>Weather History</h1>
      {error && <p style={{ color: 'red' }}>{error}</p>}

      {/* Province Dropdown */}
      <div className="dropdown-container">
        <label htmlFor="province">Select Province:</label>
        <select
          id="province"
          value={selectedProvince}
          onChange={(e) => {
            setSelectedProvince(e.target.value);
            setSelectedDistrict('');
            setWeatherHistory([]);
          }}
        >
          <option value="">-- Select Province --</option>
          {provinces.map((province, index) => (
            <option key={index} value={province[0]}>
              {province[0]}
            </option>
          ))}
        </select>
      </div>

      {/* District Dropdown */}
      <div className="dropdown-container">
        <label htmlFor="district">Select District:</label>
        <select
          id="district"
          value={selectedDistrict}
          onChange={(e) => setSelectedDistrict(e.target.value)}
          disabled={!selectedProvince}
        >
          <option value="">-- Select District --</option>
          {districts.map((district, index) => (
            <option key={index} value={district[0]}>
              {district[0]}
            </option>
          ))}
        </select>
      </div>

      {/* Weather History Table */}
      {selectedDistrict && (
        <div className="weather-history-container">
          <h2>Weather History for {selectedDistrict}</h2>
          {loadingWeather ? (
            <p>Loading...</p>
          ) : (
            <table className="weather-table">
              <thead>
                <tr>
                  <th>Date</th>
                  <th>Temperature</th>
                  <th>Humidity</th>
                  <th>Wind Speed</th>
                  <th>Gusts</th>
                  <th>Condition</th>
                  <th>Cloud Cover</th>
                  <th>UV Index</th>
                  
                </tr>
              </thead>

                {weatherHistory.map((record, index) => (

                    <td>{record[5]}</td>
                    <td>{record[6]}</td>
                    
                  </tr>
                ))}
              </tbody>
            </table>
          )}
        </div>
      )}
    </div>
  );
}

import './App.css';

function App() {
  const [provinces, setProvinces] = useState([]);
  const [selectedProvince, setSelectedProvince] = useState('');
  const [districts, setDistricts] = useState([]);
  const [selectedDistrict, setSelectedDistrict] = useState('');
  const [weatherHistory, setWeatherHistory] = useState([]);
  const [loadingWeather, setLoadingWeather] = useState(false);
  const [error, setError] = useState('');

  // Fetch provinces when the component mounts
  useEffect(() => {
    fetchProvinces();
  }, []);

  const fetchProvinces = async () => {
    try {
      const response = await fetch('http://localhost:5001/api/provinces');
      if (!response.ok) {
        throw new Error('Network response was not ok');
      }
      const data = await response.json();
      setProvinces(data);
    } catch (error) {
      setError('Error fetching provinces: ' + error.message);
    }
  };

  // Fetch districts when a province is selected
  useEffect(() => {
    if (selectedProvince) {
      fetchDistricts(selectedProvince);
    }
  }, [selectedProvince]);

  const fetchDistricts = async (province) => {
    try {
      const response = await fetch(`http://localhost:5001/api/districts/${province}`);
      if (!response.ok) {
        throw new Error('Network response was not ok');
      }
      const data = await response.json();
      setDistricts(data);
    } catch (error) {
      setError('Error fetching districts: ' + error.message);
    }
  };

  // Fetch weather history when a district is selected
  useEffect(() => {
    if (selectedDistrict) {
      fetchWeatherHistory(selectedDistrict);
    }
  }, [selectedDistrict]);

  const fetchWeatherHistory = async (district) => {
    setLoadingWeather(true);
    try {
      const response = await fetch(`http://localhost:5001/api/weather_history/${district}`);
      if (!response.ok) {
        throw new Error('Network response was not ok');
      }
      const data = await response.json();
      setWeatherHistory(data);
    } catch (error) {
      setError('Error fetching weather history: ' + error.message);
    } finally {
      setLoadingWeather(false);
    }
  };

  return (
    <div className="App">
        <a href="https://www.weatherapi.com/" title="Free Weather API">
            <img src="//cdn.weatherapi.com/v4/images/weatherapi_logo.png" alt="Weather data by WeatherAPI.com" border="0" style={{ position: 'absolute', top: '10px', left: '10px' }} />
        </a>
        <h1 style={{ marginTop: '50px' }}>Weather History</h1> 
      {error && <p style={{ color: 'red' }}>{error}</p>}

      {/* Province Dropdown */}
      <div className="dropdown-container">
        <label htmlFor="province">Select Province:</label>
        <select
          id="province"
          value={selectedProvince}
          onChange={(e) => {
            setSelectedProvince(e.target.value);
            setSelectedDistrict('');
            setWeatherHistory([]);
          }}
        >
          <option value="">-- Select Province --</option>
          {provinces.map((province, index) => (
            <option key={index} value={province[0]}>
              {province[0]}
            </option>
          ))}
        </select>
      </div>

      {/* District Dropdown */}
      <div className="dropdown-container">
        <label htmlFor="district">Select District:</label>
        <select
          id="district"
          value={selectedDistrict}
          onChange={(e) => setSelectedDistrict(e.target.value)}
          disabled={!selectedProvince}
        >
          <option value="">-- Select District --</option>
          {districts.map((district, index) => (
            <option key={index} value={district[0]}>
              {district[0]}
            </option>
          ))}
        </select>
      </div>

      {/* Weather History Table */}
      {selectedDistrict && (
        <div className="weather-history-container">
          <h2>Weather History for {selectedDistrict}</h2>
          {loadingWeather ? (
            <p>Loading...</p>
          ) : (
            <table className="weather-table">
              <thead>
                <tr>
                <th>วันที่</th>
                <th>อุณหภูมิ (°C)</th>
                <th>ความชื้น (%)</th>
                <th>ความเร็วลม (m/s)</th>
                <th>ลมกระโชก (m/s)</th>
                <th>สภาพอากาศ</th>
                <th>การปกคลุมของเมฆ (%)</th>
                <th>ดัชนี UV</th>
                  
                </tr>
              </thead>
              <tbody>
                {weatherHistory.map((record, index) => (
                  <tr key={index}>
                    <td>{record[7]}</td>
                    <td>{record[0]}</td>
                    <td>{record[1]}</td>
                    <td>{record[2]}</td>
                    <td>{record[3]}</td>
                    <td>{record[4]}</td>
                    <td>{record[5]}</td>
                    <td>{record[6]}</td>
                    
                  </tr>
                ))}
              </tbody>
            </table>
          )}
        </div>
      )}
    </div>
  );
}

export default App;

const fetchAndStoreWeatherData = async () => {
    for (const [province, subLocations] of Object.entries(locations)) {
        for (const district of subLocations) {
            try {
                // ดึงข้อมูลจาก WeatherAPI.com
                const response = await fetch(`http://api.weatherapi.com/v1/current.json?key=${API_KEY}&q=${encodeURIComponent(`${district}, ${province}`)}`);
                const weatherData = await response.json();

                // เช็คว่ามีข้อมูลหรือไม่
                if (response.ok) {
                    const { current } = weatherData;
                    const { temp_c, humidity, wind_kph, gust_kph, condition, cloud, uv } = current;

                    // ตรวจสอบข้อมูลในฐานข้อมูล
                    const query = 'SELECT * FROM weather WHERE province = $1 AND district = $2 ORDER BY created_at DESC LIMIT 1';
                    const result = await pool.query(query, [province, district]);

                    // เช็คว่ามีข้อมูลล่าสุดหรือไม่
                    if (result.rows.length > 0) {
                        const latestData = result.rows[0];
                        // เปรียบเทียบข้อมูลใหม่กับข้อมูลล่าสุดในฐานข้อมูล
                        if (
                            latestData.temperature === temp_c &&
                            latestData.humidity === humidity &&
                            latestData.wind_speed === wind_kph &&
                            latestData.gusts === gust_kph &&
                            latestData.condition === condition.text &&
                            latestData.cloud_cover === cloud &&
                            latestData.uv_index === uv
                        ) {
                            //console.log(`No changes in weather data for ${district}, ${province}. Skipping insert.`);
                            continue; // ข้ามไปยังอำเภอต่อไป
                        }
                    }

                    // ถ้าข้อมูลมีการเปลี่ยนแปลง ให้ทำการบันทึกข้อมูลใหม่
                    const insertQuery = `INSERT INTO weather (province, district, temperature, humidity, wind_speed, gusts, condition, cloud_cover, uv_index, created_at) VALUES ($1, $2, $3, $4, $5, $6, $7, $8, $9, NOW())`;
                    await pool.query(insertQuery, [
                        province,
                        district,
                        temp_c,
                        humidity,
                        wind_kph,
                        gust_kph,
                        condition.text,
                        cloud,
                        uv
                    ]);


                    const today = new Date();
                    const optionsDate = { year: 'numeric', month: 'long', day: 'numeric' };
                    const optionsTime = { hour: '2-digit', minute: '2-digit', second: '2-digit' };
                    
                    const formattedDate = today.toLocaleDateString('th-TH', optionsDate); // สำหรับวันที่
                    const formattedTime = today.toLocaleTimeString('th-TH', optionsTime); // สำหรับเวลา
                    
                    console.log(`[${formattedDate} ${formattedTime}] Weather data for ${district} in ${province} updated successfully.`);


                    //console.log(`Weather data for ${district} in ${province} updated successfully.`);

                } else {
                    console.error(`Error fetching weather data for ${district}, ${province}:`, weatherData.error);

    conn = psycopg2.connect(
        host="localhost",
        database="Weather",

    )
    return conn

@app.route('/api/provinces', methods=['GET'])
def get_provinces():
    conn = get_db_connection()
    cur = conn.cursor()
    cur.execute('SELECT DISTINCT province FROM weather order by province')
    provinces = cur.fetchall()
    cur.close()
    conn.close()
    return jsonify(provinces)

@app.route('/api/districts/<province>', methods=['GET'])
def get_districts(province):
    conn = get_db_connection()
    cur = conn.cursor()
    cur.execute('SELECT DISTINCT district FROM weather WHERE province = %s ORDER BY district', (province,))

    districts = cur.fetchall()
    cur.close()
    conn.close()
    return jsonify(districts)

@app.route('/api/weather_history/<district>', methods=['GET'])
def get_weather_history(district):
    conn = get_db_connection()
    cur = conn.cursor()
    cur.execute('''SELECT temperature, humidity, wind_speed, gusts, condition, cloud_cover, uv_index, created_at 
                   FROM weather 
                   WHERE district = %s ORDER BY created_at DESC''', (district,))
    weather_history = cur.fetchall()
    cur.close()
    conn.close()
    return jsonify(weather_history)
